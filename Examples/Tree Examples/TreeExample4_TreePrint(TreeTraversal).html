Tree Example 4: Tree Printing (illustrating Tree Traversal)
preorder_print:
void preorder_print(btNode* root, ostream& out)
// pre:  (none)
// post: values (if any) stored on tree rooted at
//       root have been inserted (each followed by
//       a space character) in preorder fashion
//       into out
{
   if (root != 0) // if empty tree, simply return 
   {
      out << root->data << ' ';
      preorder_print(root->left, out);
      preorder_print(root->right, out);
   }
}
NOTE: The base case (situation: empty tree; action: return) is implicit as commented.
inorder_print:
void inorder_print(btNode* root, ostream& out)
// pre:  (none)
// post: values (if any) stored on tree rooted at
//       root have been inserted (each followed by
//       a space character) in inorder fashion
//       into out
{
   if (root != 0) // if empty tree, simply return 
   {
      inorder_print(root->left, out);
      out << root->data << ' ';
      inorder_print(root->right, out);
   }
}
NOTE: The base case (situation: empty tree; action: return) is implicit as commented.
postorder_print:
void postorder_print(btNode* root, ostream& out)
// pre:  (none)
// post: values (if any) stored on tree rooted at
//       root have been inserted (each followed by
//       a space character) in postorder fashion
//       into out
{
   if (root != 0) // if empty tree, simply return 
   {
      postorder_print(root->left, out);
      postorder_print(root->right, out);
      out << root->data << ' ';
   }
}
NOTE: The base case (situation: empty tree; action: return) is implicit as commented.
backward_inorder_print:
void backward_inorder_print(btNode* root, ostream& out, int depth)
// pre:  (none)
// post: values (if any) stored on tree rooted at
//       root have been inserted into out to produce an output
//       illustrated below:
//       tree:     3         output:     5
//                / \                3
//               2   5                       4
//              / \                      2
//             1   4                             0
//              \                            1
//               0
{
   if (root != 0) // if empty tree, simply return 
   {
      backward_inorder_print(root->right, out, depth + 1);
      out << setw(4*depth) << "";
      out << root->data << endl;
      backward_inorder_print(root->left, out, depth + 1);
   }
}
NOTE: The base case (situation: empty tree; action: return) is implicit as commented.
