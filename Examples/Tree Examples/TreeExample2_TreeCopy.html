Tree Example 2: tree_copy
One way:
void tree_copy(btNode*& dst_root, btNode* src_root)
// pre:  destination root pointer (dst_root) does not
//       point to anything relevant
// post: dst_root has been made the root to a new
//       tree that is identical (wrt structure and
//       values) to the tree rooted at src_root
{
   if (src_root == 0) // base case
      dst_root = 0;
   else
   {
      dst_root = new btNode;
      dst_root->data = src_root->data;
      tree_copy(dst_root->left, src_root->left);
      tree_copy(dst_root->right, src_root->right);
   }
}
QUIZ: Would the leaf nodes of the copy (tree rooted at dst_root) properly null terminated?
Another way:
btNode* tree_copy(btNode* src_root)
// pre:  (none)
// post: a root pointer to a new tree that is identical
//       (wrt structure and values) to the tree rooted
//       at src_root is returned
{
   if (src_root == 0) // base case
      return 0;
   else
   {
      btNode* ret_root = new btNode;
      ret_root->data = src_root->data;
      ret_root->left = tree_copy(src_root->left);
      ret_root->right = tree_copy(src_root->right);
      return ret_root;
   }
}
