Tree Example 3: Tree Inspection
size:
int size(btNode* root)
// pre:  (none)
// post: number of nodes contained in tree rooted at root
//       is returned
{
   if (root == 0) // base case
      return 0;
   return 1 + size(root->left) + size(root->right);
}
depth:
int depth(btNode* root)
// pre:  (none)
// post: depth of tree rooted at root is returned
{
   if (root == 0) // base case
      return 0;
   int depthLST = depth(root->left),
       depthRST = depth(root->right);
   return (depthLST > depthRST) ? 1 + depthLST : 1 + depthRST;
}
NOTE: By the definition used here, depth of an empty tree is 0.
wtBalanced:
bool wtBalanced(btNode* root)
// pre:  (none)
// post: true is returned if tree rooted at root is
//       weight-balanced, otherwiese false is returned
{
   if (root == 0) // base case
      return true;
   int size_diff = size(root->left) - size(root->right);
   if (size_diff < -1 || size_diff > 1) return false;
   return wtBalanced(root->left) && wtBalanced(root->right);
}
full: (per textbook, where tree is considered full if every leaf has same depth and every non-leaf has two non-empty children)  (empty tree is full by definition)
bool full(btNode* root)
// pre:  (none)
// post: true is returned if tree rooted at root is
//       a full tree, otherwiese false is returned
{
   if (root == 0) // empty
      return true;
   return chkFull(root, 0) != -1;
}
/////////////////////////////////////////////////////////////
int chkFull(btNode* btNodePtr, int depRef)
// pre:  btNodePtr != 0
// post: returns -1 if depth discrepancy found
{
   ++depRef;
   if (btNodePtr->left == 0 && btNodePtr->right == 0) // leaf
      return depRef;
   if (btNodePtr->left == 0 && btNodePtr->right != 0
       ||
       btNodePtr->left != 0 && btNodePtr->right == 0)
      return -1;
   int depRefLeft  = chkFull(btNodePtr->left,  depRef),
       depRefRight = chkFull(btNodePtr->right, depRef);
   if ( (depRefLeft == depRefRight) && (depRefLeft != -1) )
      return ++depRef;
   return -1;
}
full: (per certain others, where tree is considered full if every node has either zero or two non-empty children) (empty tree is full by definition)
bool full(btNode* root)
// pre:  (none)
// post: true is returned if tree rooted at root is
//       a full tree, otherwiese false is returned
{
   if (root == 0) // base case
      return true;
   if (root->left == 0 && root->right != 0
       ||
       root->left != 0 && root->right == 0)
      return false;
   return full(root->left) && full(root->right);
}
