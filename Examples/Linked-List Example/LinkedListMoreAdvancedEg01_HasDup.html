In-class Example: Function to determine if a given linked list has at least one node containing a data item equal to that of another node.
By definition:
An empty list cannot have duplicates.
Code:
bool HasDup(Node* head)
{
   if (head == 0 || head->link == 0) return false;
   Node* cursor = 0;
   int anchorData;
   while (head->link != 0) // anchor node not last node
   {
      anchorData = head->data;
      cursor = head->link; // points to 1st node downstream of anchor node
      while (cursor != 0)
      {
         if (cursor->data == anchorData) return true;
         cursor = cursor->link;
      }
      head = head->link;
   }
   return false;
}
