int main()
{
   int *Process(int *intPtr, int size);
   int intArray[5] = {10, 20, 30, 40, 50}, *intPtr = 0;
   intPtr = Process(intArray, 5);
   cout << "Desired value is " << *intPtr;
   return 0;
}

int *Process(int *intPtr, int size)
{
   int i, desiredValue = *intPtr, iDesiredValue = 0;
   for (i = 1; i < size; i++)
      if ( *(intPtr + i) > desiredValue )
      {
         desiredValue = *(intPtr + i);
         iDesiredValue = i;
      }
   return (intPtr + iDesiredValue);
}
