#include <iostream>
#include <cstdlib>
using namespace std;

void f1(int* xPtr);

int main()
{
   int x = 12;
   cout << "x = " << x << endl;
   f1(&x);
   cout << "x = " << x << endl;
   return EXIT_SUCCESS;
}

void f1(int* xPtr)
{
   *xPtr = 34;

}
